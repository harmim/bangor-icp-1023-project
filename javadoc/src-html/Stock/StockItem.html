001package Stock;
002
003
004/**
005 * Representation of item in stock.
006 *
007 * @author Dominik Harmim harmim6@gmail.com
008 */
009public class StockItem
010{
011        /**
012         * Item ID. Five alpha-numeric characters.
013         */
014        private String itemId;
015
016        /**
017         * Item description.
018         */
019        private String itemDesc;
020
021        /**
022         * Item price in pounds sterling.
023         */
024        private double price;
025
026        /**
027         * Quantity of item in stock.
028         */
029        private int quantity;
030
031        /**
032         * Level at which to re-order.
033         */
034        private int reOrderLevel;
035
036
037        /**
038         * Creates stock item and initializes it's instance variables.
039         *
040         * @param itemId item ID
041         * @param itemDesc item description
042         * @param price item price in pounds
043         * @param quantity quantity of item in stock
044         * @param reOrderLevel level at which to re-order
045         */
046        public StockItem(String itemId, String itemDesc, double price, int quantity, int reOrderLevel)
047        {
048                this.itemId = itemId;
049                this.itemDesc = itemDesc;
050                this.price = price;
051                this.quantity = quantity;
052                this.reOrderLevel = reOrderLevel;
053        }
054
055
056        /**
057         * Returns item ID.
058         *
059         * @return item ID
060         */
061        public String getItemId()
062        {
063                return itemId;
064        }
065
066
067        /**
068         * Returns item description.
069         *
070         * @return item description
071         */
072        public String getItemDesc()
073        {
074                return itemDesc;
075        }
076
077
078        /**
079         * Returns item price in pounds.
080         *
081         * @return item price in pounds
082         */
083        public double getPrice()
084        {
085                return price;
086        }
087
088
089        /**
090         * Sets item price.
091         *
092         * @param price item price in pounds
093         */
094        public void setPrice(double price)
095        {
096                this.price = price;
097        }
098
099
100        /**
101         * Returns item quantity in stock.
102         *
103         * @return item quantity in stock
104         */
105        public int getQuantity()
106        {
107                return quantity;
108        }
109
110
111        /**
112         * Sets item quantity in stock.
113         *
114         * @param quantity item quantity in stock
115         */
116        public void setQuantity(int quantity)
117        {
118                this.quantity = quantity;
119        }
120
121
122        /**
123         * Returns level at which to re-order.
124         *
125         * @return level at which to re-order
126         */
127        public int getReOrderLevel()
128        {
129                return reOrderLevel;
130        }
131
132
133        /**
134         * Sets level at which to re-order.
135         *
136         * @param reOrderLevel level at which to re-order
137         */
138        public void setReOrderLevel(int reOrderLevel)
139        {
140                this.reOrderLevel = reOrderLevel;
141        }
142
143
144        /**
145         * Returns formatted string representation of stock item.
146         *
147         * @return formatted string representation of stock item
148         */
149        public String format()
150        {
151                return String.format("%-5s %-20s %10.4f %8d %8d", itemId, itemDesc, price, quantity, reOrderLevel);
152        }
153
154
155        /**
156         * Returns string representation of stock item.
157         *
158         * @return string representation of stock item
159         */
160        @Override
161        public String toString()
162        {
163                return getClass().getName() + '{'
164                        + "itemId='" + itemId + '\''
165                        + ", itemDesc='" + itemDesc + '\''
166                        + ", price=" + price
167                        + ", quantity=" + quantity
168                        + ", reOrderLevel=" + reOrderLevel
169                        + '}';
170        }
171}
